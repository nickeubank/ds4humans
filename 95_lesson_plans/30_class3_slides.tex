\begin{frame}[c]{The Big Picture}
\begin{itemize}
    \pause \item Our goal: Estimate \alert{Average Treatment Effect (ATE)}
    \pause \item We can \alert{measure}: \\[1ex] $\widehat{ATE} = E(Y^1|D=1) - E(Y^0|D=0)$
    \pause \item If \alert{no baseline differences}: \\[1ex] $\widehat{ATE} = ATT$
    \pause \item If also \alert{no differential treatment effects}: \\[1ex]
    $\widehat{ATE} = ATT = ATE$ {\emoji ðŸ¥³}
\end{itemize}    
\end{frame}

\begin{frame}[c]{Why do we care about ATE?}
We want to answer causal questions because \alert{we are considering taking an action at scale}, and want to know the likely effect. \\
\vspace{0.5cm}
\pause 
In our study, we will have some people treated and some untreated. \\
\vspace{0.5cm}
\pause 
But if we take the action being considered, \alert{we will do it for everyone.}
\begin{itemize}
    \item Authorize a new drug.
    \item Change the price of a product.
    \item Change webpage design.
\end{itemize}
\pause Average effect will be ATE.
\end{frame}

\begin{frame}[c]{No Baseline Differences}
\vspace{-1cm}
    $$E(Y^0|D=1) - E(Y^0|D=0) = 0$$
\alert{In a world without treatment}, outcomes Y are the same (on average) for both groups. \\
\vspace{0.5cm}
\begin{itemize}
    \pause \item Does \alert{NOT} require no differences between $D=0$ and $D=1$, 
    \pause \item Does \alert{NOT} require no differences in $Y$ across individuals.
\end{itemize}
\vspace{0.5cm}
\pause Requires \alert{no average differences in outcome Y between $D=0$ and $D=1$}.
\end{frame}


\begin{frame}[c]{No Differential Treatment Effects}
\vspace{-1cm}
    $$E(Y^1|D=1) - E(Y^0|D=1) =  E(Y^1|D=0) - E(Y^0|D=0)$$
The effect of $T$ is the same (on average) for both groups. \\
\vspace{0.5cm}
\begin{itemize}
    \pause \item If not met, but no baseline differences is met, still get \emph{a} treatment effect: ATT.
\end{itemize}
\end{frame}
